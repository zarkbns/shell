Integration guide

1 Install

Terminal 
npm install @shellxyz/sdk

2 Initialize inside your wallet client

setup.ts 
import ShellSDK from '@shellxyz/sdk'

ShellSDK.init({
  contactList: walletClient.getSavedAddresses(),
  solThreshold: 10
})

3 Intercept transactions before signing

interceptor.ts 
async function onTransactionRequest(tx) {
  const result = await ShellSDK.analyzeTransaction({
    address: tx.destination,
    transferAmount: tx.amount,
    transactionType: tx.type
  })

  if (result.blocked) return wallet.reject(result.reason)
  if (result.requiresPIN) return wallet.triggerPIN()
  return wallet.proceedToSign(tx)
}

4 Deep decode — read what the transaction actually does

decoder.ts 
const decoded = await ShellSDK.decodeAndAnalyze({
  serializedTransaction: tx.serialized,
  encoding: 'base64',
  transferAmount: tx.amount,
  contactList: wallet.contacts
})

if (decoded.containsUnknownPrograms) {
  wallet.showWarning(decoded.warning)
}

5 Away Mode — block background signing

awayMode.ts 
// User locks their screen or goes idle
ShellSDK.setAwayMode(true)

// Any background signing attempt is blocked instantly
// Re-enable when user returns
ShellSDK.setAwayMode(false)
